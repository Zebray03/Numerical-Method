```matlab
function [x,n0,numerical_error]=jacobi(A,b,n,eps,error_rule,norm_type,x0)
% 用Jacobi方法解线性方程组Ax=b
% 最大迭代次数为n
% 用户指标为eps
% error_rule为停机准则
%   remaining为残量准则
%   adjacent为相邻误差准则
%   backward为后验误差准则
%   error为真实误差准则
% norm_type为范数种类
%   1为1-范数
%   inf为无穷范数
% 返回解向量x
% 最终迭代次数n0
% 真实数值误差numerical_error
if(~exist('x0','var'))
    x0=zeros(size(A,1),1);
end
x=zeros(size(A,1),1);
D=zeros(size(A,1),size(A,1));
x_star=ones(size(A,1));
for i=1:size(A,1)
    D(i,i)=1/A(i,i);
end
G=eye(size(A,1))-D*A;
g=D*b;
if (error_rule=="remaining")
    for i=1:n
        x=G*x0+g;
        if(norm_type=="1")
            if(norm(A*x-b,1)<eps)
                n0=i;
                numerical_error=norm(x-x_star,1);
                return
            else
                x0=x;
            end
        elseif(norm_type=="2")
            if(norm(A*x-b)<eps)
                n0=i;
                numerical_error=norm(x-x_star);
                return
             else
                x0=x;
            end
        elseif(norm_type=="inf")
            if(norm(A*x-b,"inf")<eps)
                n0=i;
                numerical_error=norm(x-x_star,"inf");
                return
            else
                x0=x;
            end
        else
            quit
        end
    end
    n0=n;
    if(norm_type=="1")
        numerical_error=norm(x-x_star,1);
    elseif(norm_type=="2")
        numerical_error=norm(x-x_star);
    elseif(norm_type=="inf")
        numerical_error=norm(x-x_star,"inf");
    else 
        quit
    end
elseif(error_rule=="adjacent")
    for i=1:n
        x=G*x0+g;
        if(norm_type=="1")
            if(norm(x-x0,1)<eps)
                n0=i;
                numerical_error=norm(x-x_star,1);
                return
            else
                x0=x;
            end
        elseif(norm_type=="inf")
            if(norm(x-x0,"inf")<eps)
                n0=i;
                numerical_error=norm(x-x_star,"inf");
                return
            else
                x0=x;
            end
        else
            quit
        end
    end
    n0=n;
    if(norm_type=="1")
        numerical_error=norm(x-x_star,1);
    elseif(norm_type=="inf")
        numerical_error=norm(x-x_star,"inf");
    else 
        quit
    end
elseif(error_rule=="backward")
    delta1=0;
    delta2=0;
    for i=1:n
        x=G*x0+g;
        if(norm_type=="1")
            if(i==1)
                delta1=norm(x-x0,1);
            else
                delta2=norm(x-x0,1);
                if(delta2^2/(delta1-delta2)<eps)
                    n0=i;
                    numerical_error=norm(x-x_star,1);
                    return
                else
                    x0=x;
                    delta1=delta2;
                end
            end
        elseif(norm_type=="inf")
            if(i==1)
                delta1=norm(x-x0,"inf");
            else
                delta2=norm(x-x0,"inf");
                if(delta2^2/(delta1-delta2)<eps)
                    n0=i;
                    numerical_error=norm(x-x_star,"inf");
                    return
                else
                    x0=x;
                    delta1=delta2;
                end
            end
        end
    end
    n0=n;
    if(norm_type=="1")
        numerical_error=norm(x-x_star,1);
    elseif(norm_type=="inf")
        numerical_error=norm(x-x_star,"inf");
    else 
        quit
    end
elseif(error_rule=="error")
    for i=1:n
        x=G*x0+g;
        if(norm_type=="1")
            if(norm(x-x_star,1)<eps)
                n0=i;
                numerical_error=norm(x-x_star,1);
                return
            else
                x0=x;
            end
        elseif(norm_type=="2")
            if(norm(x-x_star)<eps)
                n0=i;
                numerical_error=norm(x-x_star);
                return
            else
                x0=x;
            end
        elseif(norm_type=="inf")
            if(norm(x-x_star,"inf")<eps)
                n0=i;
                numerical_error=norm(x-x_star,"inf");
                return
            else
                x0=x;
            end
        else
            quit
        end
    end
    n0=n;
    if(norm_type=="1")
        numerical_error=norm(x-x_star,1);
    elseif(norm_type=="inf")
        numerical_error=norm(x-x_star,"inf");
    else 
        quit
    end
else
    quit
end
```

```matlab
function [x,n0,numerical_error]=sor(A,b,n,omega,eps,x0)
% 用SOR方法解线性方程组Ax=b
% 最大迭代次数为n
% 松弛因子为omega
% 用户指标为eps
% 返回解向量x
% 最终迭代次数n0
% 数值误差numerical_error
if(~exist('x0','var'))
    x0=zeros(size(A,1),1);
end
x=zeros(size(A,1),1);
x_star=ones(size(A,1));
for i=1:n
    for j=1:size(A,1)
        x(j)=(1-omega)*x0(j)+omega/A(j,j)*(b(j)-A(j,1:j-1)*x(1:j-1)-A(j,j+1:size(A,1))*x0(j+1:size(A,1)));
    end
    if(norm(x-x_star)<eps)
        n0=i;
        numerical_error=norm(x-x_star);
        return
    else
        x0=x;
    end
end
n0=n;
numerical_error=norm(x-x_star);
```

```matlab
function [x,n0,numerical_error]=half_iteration_jacobi(A,b,n,eps,m,y0)
% 用半迭代加速的Jacobi方法解线性方程组Ax=b
% 最大迭代次数为n
% 用户指标为eps
% 循环指标为m
% 初始向量为y0（默认为0）
% 返回解向量x
% 最终迭代次数n0
% 真实数值误差numerical_error
if(~exist('x0','var'))
    y0=zeros(size(A,1),1);
end
y1=zeros(size(A,1),1);
y2=zeros(size(A,1),1);
D=zeros(size(A,1),size(A,1));
x_star=ones(size(A,1));
for i=1:size(A,1)
    D(i,i)=1/A(i,i);
end
G=eye(size(A,1))-D*A;
g=D*b;
lambda_max=max(eig(G));
lambda_min=min(eig(G));
v=2/(2-lambda_max-lambda_min);
ksi=(2-lambda_max-lambda_min)/(lambda_max-lambda_min);
rho=2;
y1=G*y0+g;
if(norm(A*y1-b)<eps)
    x=y1;
    n0=1;
    numerical_error=norm(x-x_star);
    return
else
    for i=1:min([m n])
        rho=1/(1-rho/(4*ksi*ksi));
        y2=rho.*(v.*(G*y1+g)+(1-v).*y1)+(1-rho).*y0;
        if(norm(A*y2-b)<eps)
            x=y2;
            n0=i+1;
            numerical_error=norm(x-x_star);
            return
        end
        y0=y1;
        y1=y2;
    end
end
x=y2;
n0=min([m n])+1;
numerical_error=norm(x-x_star);
```

```matlab
function [x,n_list,numerical_error_list,remaining_list]=cg(A,b,n,eps,x0)
% 用CG方法解线性方程组Ax=b
% 最大迭代次数为n
% 用户指标为eps
% 返回解向量x
% 迭代次数列表n_list
% 真实数值误差列表numerical_error_list
% 残量列表remaining_list
if(~exist('x0','var'))
    x0=zeros(size(A,1),1);
end
x_star=ones(size(A,1));
n_list=zeros(n,1);
numerical_error_list=zeros(n,1);
remaining_list=zeros(n,1);
r=A*x0-b;
p=-r;
k1=r'*r;
for i=1:n
    n_list(i)=i;
    q=A*p;
    k2=p'*q;
    alpha=k1/k2;
    x=x0+alpha*p;
    if(norm(A*x-b)<eps)
        numerical_error_list(i)=norm(x-x_star);
        remaining_list(i)=norm(A*x-b);
        break
    else
        numerical_error_list(i)=norm(x-x_star);
        remaining_list(i)=norm(A*x-b);
        x0=x;
        r=r+alpha*q;
        k3=r'*r;
        beta=k3/k1;
        p=-r+beta*p;
        k1=k3;
    end
end
flag=find(n_list==0)-1;
n_list=n_list(1:flag);
numerical_error_list=numerical_error_list(1:flag);
remaining_list=remaining_list(1:flag);
```
