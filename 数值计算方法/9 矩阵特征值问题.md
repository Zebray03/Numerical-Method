```matlab
function [lambda,x,lambda_list,x_list]=power_method(A,v,eps)
if ~exist('eps','var')
    eps=1e-8;
end
if ~exist('v','var')
    v=ones(size(A,1),1);
end
u=v;
lambda_list=0;
x_list=zeros(size(A,1),1);
while true
    lambda=0;
    v=A*v;
    for i=1:size(v,1)
        if abs(v(i))>abs(lambda)
            lambda=v(i);
        end
    end
    v=v./lambda;
    x_test=max(abs(u-v));
    if(x_test<eps)
        break
    else
        lambda_list=[lambda_list lambda];
        x_list=[x_list v];
        u=v;
    end
end
x=v/norm(v);
lambda_list=lambda_list(:,2:end);
x_list=x_list(:,2:end);
```

```matlab
            lambda=v(i);

    if(x_test<eps)
        break
    else
        lambda_list=[lambda_list lambda_atikens];
        x_list=[x_list v];
        u=v;
        b0=b1;
        b1=lambda;
    end
end
x=v/norm(v);
lambda_list=lambda_list(:,2:end);
x_list=x_list(:,2:end);
```

```matlab
function [lambda,x,lambda_list,x_list]=inverse_power_method(A,q,v,eps)

    q=0;
end
if ~exist('eps','var')
    eps=1e-8;
end
if ~exist('v','var')
    v=ones(size(A,1),1);

    for i=1:size(v,1)

            lambda=v(i);

    else

        x_list(:,size(x_list,2)+1)=v./norm(v);
        u=v;
    end
```

```matlab
function [lambda1,lambda2,x1,x2]=iterate_meanwhile_method(A,eps)
if ~exist('eps','var')
    eps=1e-8;
end
n=size(A,1);
% 初始直交阵选取
v=rand(n,1);
b=sum(v.*v)/2;
V=eye(n)-v*v'./b;
index=sort(randperm(n,2));
for i=1:2
    V(i)=V(index(i));
end
V=V(:,1:2);

U=A*V;
B=V'*U;
[ve,d]=eig(B);
lambda1=d(2,2);
lambda2=d(1,1);
x1=ve(:,2);
x2=ve(:,1);
x1=x1./norm(x1);
x2=x2./norm(x2);
W=[x1 x2];
[V,~]=mgs(U*W);

U=A*V;
B=V'*U;
[ve,d]=eig(B);
temp1=d(2,2);
temp2=d(1,1);
while abs(lambda1-temp1)>=eps || abs(lambda2-temp2)>=eps
    lambda1=temp1;
    lambda2=temp2;
    x1=ve(:,2);
    x2=ve(:,1);
    x1=x1./norm(x1);
    x2=x2./norm(x2);
    W=[x1 x2];
    [V,~]=mgs(U*W);

    U=A*V;
    B=V'*U;
    [ve,d]=eig(B);
    temp1=d(2,2);
    temp2=d(1,1);
end
x1=V(:,1);
x2=V(:,2);
```

```matlab
function [lambda,e_list,d_list]=classical_jacobi(A,eps)
if ~exist('eps','var')
    eps=1e-8;
end
pivot=0;
e=0;
for i=1:size(A,1)
    for j=i+1:size(A,2)
        if abs(A(i,j))>=eps*sqrt(abs(A(i,i)*A(j,j)))
            e=e+2*A(i,j)^2;
        end
        if abs(A(i,j))>pivot
            pivot=abs(A(i,j));
            index=[i,j];
        end
    end
end
e_list=e;
d=diag(A);
d_list=0;
while e>=eps
    if abs(A(index(1),index(1))-A(index(2),index(2)))<eps
        c=1/sqrt(2);
        t=sign(A(index(1),index(2)));
        s=t/sqrt(2);
    else
        t=(A(index(1),index(1))-A(index(2),index(2)))/(2*A(index(1),index(2)));
        t=sign(t)/(abs(t)+sqrt(1+t^2));
        c=1/sqrt(1+t^2);
        s=c*t;
    end
    col_p=c*A(:,index(1))+s*A(:,index(2));
    col_q=-s*A(:,index(1))+c*A(:,index(2));
    row_p=c*A(index(1),:)+s*A(index(2),:);
    row_q=-s*A(index(1),:)+c*A(index(2),:);
    temp_p=A(index(1),index(1))+t*A(index(1),index(2));
    temp_q=A(index(2),index(2))-t*A(index(1),index(2));
    A(:,index(1))=col_p;
    A(:,index(2))=col_q;
    A(index(1),:)=row_p;
    A(index(2),:)=row_q;
    A(index(1),index(1))=temp_p;
    A(index(2),index(2))=temp_q;
    A(index(1),index(2))=0;
    A(index(2),index(1))=0;
    pivot=0;
    e=0;
    for i=1:size(A,1)
        for j=i+1:size(A,2)
            if abs(A(i,j))>=eps*sqrt(abs(A(i,i)*A(j,j)))
                e=e+2*A(i,j)^2;
            end
            if abs(A(i,j))>pivot
                pivot=abs(A(i,j));
                index=[i,j];
            end
        end
    end
    e_list=[e_list e];
    d1=diag(A);
    d=d1-d;
    sum_d=0;
    for i=1:size(A,1)
        sum_d=sum_d+d(i)^2;
    end
    d_list=[d_list sum_d];
    d=d1;
end
lambda=diag(A);
d_list=d_list(2:end);
```

```matlab
function [lambda,e_list,d_list]=threshold_jacobi(A,eps)

    eps=1e-8;
end

    for i=1:size(A,1)

end
```

```matlab
    left="open";

end
if right=="open"

s_a=0;
s_b=0;

beta_list=zeros(n-1,1);
for i=1:n-1
    alpha_list(i)=A(i,i);
    beta_list(i)=A(i,i+1);
end
alpha_list(n)=A(n,n);
list=zeros(n+1,1);
for i=1:n+1
    if i==1
        list(i)=1;
    elseif i==2
        list(i)=alpha_list(1)-a;
    else
        list(i)=(alpha_list(i-1)-a)*list(i-1)-beta_list(i-2)^2*list(i-2);

i=1;
while true

        if temp~=list(i)
            count=count+1;
            temp=list(i);
        end
    end
    while i<n+1
        i=i+1;
        if temp==list(i)
            count=count+1;
        else
            break
        end
    end

    if i==n+1

    end
```
